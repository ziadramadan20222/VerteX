const jobs = [
  {name:"Wirinix - Appointment Setter", location:"WFH", english:["B2","C1"], expWeight:1.2},
  {name:"Launch Pad - Sales Agent", location:"Sheikh Zayed", english:["C1"], expWeight:1.4},
  {name:"ACES - Telesales", location:"Abbassia", english:["B2"], expWeight:1.1},
  {name:"TTC - Telesales", location:"Nasr City", english:["B2","C1"], expWeight:1.3},
  {name:"TTP - Telesales", location:"Sheraton", english:["B2","C1"], expWeight:1.3},
  {name:"REZ - Cold Caller", location:"WFH", english:["B2","C1"], expWeight:1.5},
  {name:"Nexaura - Cold Calling", location:"Gisr El Suez", english:["B2","C1","C2"], expWeight:1.4},
  {name:"LeadBull - Telesales", location:"Zahraa El Maadi", english:["B2","C1"], expWeight:1.6},
  {name:"Symbios - BDR", location:"Nasr City", english:["C1","C2"], expWeight:1.8}
];

function getEnglishScore(level){
  if(level=="B1") return 1;
  if(level=="B2") return 2;
  if(level=="C1") return 3;
  if(level=="C2") return 4;
  return 0;
}

function createCard(name, score){
  return `<div class='card'><h3>${name}</h3><p>${score}% AI Match</p></div>`;
}

function findJobs(){

  let age=parseInt(document.getElementById('age').value);
  let english=document.getElementById('english').value;
  let location=document.getElementById('location').value;
  let experience=document.getElementById('experience').value;
  let graduation=document.getElementById('graduation').value;
  let nationality=document.getElementById('nationality').value;

  let result=document.getElementById('results');
  result.innerHTML="";

  if(!age || age<18){
    alert("Must be 18+");
    return;
  }

  let userEng=getEnglishScore(english);
  let userExp=parseFloat(experience)||0;

  jobs.forEach(job=>{

    let score=0;

    // AI weighted scoring system
    if(job.location===location) score+=20;

    // English AI match
    let jobEng=Math.max(...job.english.map(getEnglishScore));
    score+= (userEng/jobEng)*25;

    // Experience weight AI
    score+= userExp * job.expWeight * 10;

    // Graduation AI bonus
    if(graduation=="Grad") score+=10;
    else if(graduation=="Undergrad") score+=7;
    else score+=5;

    // Nationality soft match
    if(nationality=="Egyptian") score+=8;
    else score+=6;

    // Age compatibility AI curve
    let ageScore=age<=30?10:5;
    score+=ageScore;

    score=Math.min(100,Math.round(score));

    if(score>=50){
      result.innerHTML+=createCard(job.name,score);
    }
  });
}
